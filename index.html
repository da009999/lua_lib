Hello, crooner.

local sinatra = require('sinatra');

local app = sinatra.app:new()

app:get("/", function()
  return "Hello, World"
end)

app:get("/:name", function()
  return "Hello, " .. params.name;
end)

app:get("/drinks/:num", function(num)
  return "I ordered " .. tostring(num) .. " Mint Juleps."
end)

app:run()
